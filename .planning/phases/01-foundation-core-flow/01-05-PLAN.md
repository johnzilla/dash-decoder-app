export const SYSTEM_PROMPT = `You are an expert automotive diagnostic assistant with deep knowledge of:
- Dashboard warning lights for all vehicle makes 1996 and newer (OBD-II era)
- OBD-II diagnostic trouble codes
- Vehicle identification from dashboard styling

Your responses must be in valid JSON format. Be direct and confident in your analysis.
If you cannot identify something with confidence, indicate that clearly.`;

export function getAnalysisPrompt(): string {
  return `Analyze this dashboard photo and identify:

1. The warning light visible (name and any visible error code)
2. Your confidence level (0-1) in the warning light identification
3. The vehicle make, model, and approximate year based on dashboard styling
4. Your confidence level (0-1) in the vehicle guess

Respond with ONLY valid JSON in this exact format:
{
  "warningLight": {
    "name": "Warning light name",
    "code": "P0420 or null if not visible",
    "confidence": 0.85
  },
  "vehicleGuess": {
    "make": "Toyota or null if unknown",
    "model": "Camry or null if unknown",
    "year": 2020 or null if unknown,
    "confidence": 0.7
  }
}`;
}

export function getDiagnosisPrompt(
  warningLight: string,
  vehicle: { make: string; model: string; year: number }

import {
  VisionAnalysisResult,
  VisionAnalysisResultSchema,
  WarningLight,
  Severity,
  FixStep,
} from '@/types';
import { SYSTEM_PROMPT, getAnalysisPrompt, getDiagnosisPrompt } from './prompts';

interface OpenAIError {
  error: {
    message: string;
    type: string;
    code: string;
  };
}

/**
 * Analyzes a dashboard photo to identify warning light and guess vehicle
 */
export async function analyzeWarningLight(
  imageDataUrl: string
): Promise<VisionAnalysisResult> {
  const apiKey = import.meta.env.VITE_OPENAI_API_KEY;

  if (!apiKey) {
    throw new Error('OpenAI API key not configured. Please add VITE_OPENAI_API_KEY to your .env file.');
  }

  const response = await fetch('https://api.openai.com/v1/chat/completions', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      Authorization: `Bearer ${apiKey}`,
    },
    body: JSON.stringify({
      model: 'gpt-4o',
      messages: [
        {
          role: 'system',
          content: SYSTEM_PROMPT,
        },
        {
          role: 'user',
          content: [
            {
              type: 'text',
              text: getAnalysisPrompt(),
            },
            {
              type: 'image_url',
              image_url: {
                url: imageDataUrl,
                detail: 'high', // High detail for better accuracy
              },
            },
          ],
        },
      ],
      max_tokens: 500,
      temperature: 0.1, // Low temperature for consistent output
    }),
  });

  if (!response.ok) {
    const errorData = (await response.json()) as OpenAIError;
    throw new Error(
      errorData.error?.message || `Vision API error: ${response.statusText}`
    );
  }

  const data = await response.json();
  const content = data.choices[0]?.message?.content;

  if (!content) {
    throw new Error('No response from Vision API');
  }

  try {
    // Parse JSON from response
    const jsonMatch = content.match(/\{[\s\S]*\}/);
    if (!jsonMatch) {
      throw new Error('Could not find JSON in response');
    }

    const parsed = JSON.parse(jsonMatch[0]);

    // Validate with Zod schema
    const validated = VisionAnalysisResultSchema.parse({
      warningLight: {
        name: parsed.warningLight?.name || 'Unknown warning light',
        code: parsed.warningLight?.code || null,
        confidence: parsed.warningLight?.confidence || 0.5,
      },
      vehicleGuess: {
        make: parsed.vehicleGuess?.make || null,
        model: parsed.vehicleGuess?.model || null,
        year: parsed.vehicleGuess?.year || null,
        confidence: parsed.vehicleGuess?.confidence || 0,
      },
      rawAnalysis: content,
    });

    return validated;
  } catch (parseError) {
    console.error('Failed to parse Vision API response:', content);
    throw new Error(
      'Failed to parse AI response. Please try again with a clearer photo.'
    );
  }
}

interface DiagnosisGenerationResult {
  severity: Severity;
  explanation: string;
  safetyGuidance: string;
  fixSteps: FixStep[];
}

/**
 * Generates a complete diagnosis for a confirmed warning light and vehicle
 */
export async function generateDiagnosis(
  warningLight: WarningLight,
  vehicle: { make: string; model: string; year: number }
): Promise<DiagnosisGenerationResult> {
  const apiKey = import.meta.env.VITE_OPENAI_API_KEY;

  if (!apiKey) {
    throw new Error('OpenAI API key not configured.');
  }

  const response = await fetch('https://api.openai.com/v1/chat/completions', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      Authorization: `Bearer ${apiKey}`,
    },
    body: JSON.stringify({
      model: 'gpt-4o',
      messages: [
        {
          role: 'system',
          content: SYSTEM_PROMPT,
        },
        {
          role: 'user',
          content: getDiagnosisPrompt(
            `${warningLight.name}${warningLight.code ? ` (${warningLight.code})` : ''}`,
            vehicle
          ),
        },
      ],
      max_tokens: 1000,
      temperature: 0.3, // Slightly higher for more varied fix instructions
    }),
  });

  if (!response.ok) {
    const errorData = (await response.json()) as OpenAIError;
    throw new Error(
      errorData.error?.message || `API error: ${response.statusText}`
    );
  }

  const data = await response.json();
  const content = data.choices[0]?.message?.content;

  if (!content) {
    throw new Error('No response from API');
  }

  try {
    const jsonMatch = content.match(/\{[\s\S]*\}/);
    if (!jsonMatch) {
      throw new Error('Could not find JSON in response');
    }

    const parsed = JSON.parse(jsonMatch[0]);

    return {
      severity: parsed.severity || 'warning',
      explanation: parsed.explanation || 'Unable to provide explanation.',
      safetyGuidance: parsed.safetyGuidance || 'Consult a mechanic if concerned.',
      fixSteps: (parsed.fixSteps || []).map((step: { step: number; instruction: string; notes?: string }, idx: number) => ({
        step: step.step || idx + 1,
        instruction: step.instruction || 'Step details unavailable',
        notes: step.notes || undefined,
      })),
    };
  } catch (parseError) {
    console.error('Failed to parse diagnosis response:', content);
    throw new Error('Failed to generate diagnosis. Please try again.');
  }
}

```typescript
import { useState, useCallback } from 'react';
import type { VisionAnalysisResult } from '@/types';
import { analyzeWarningLight } from '@/lib/api/openai';

interface UseVisionAnalysisReturn {
  analyze: (imageDataUrl: string) => Promise<VisionAnalysisResult | null>;
  isAnalyzing: boolean;
  error: string | null;
  result: VisionAnalysisResult | null;
  reset: () => void;
}

export function useVisionAnalysis(): UseVisionAnalysisReturn {
  const [isAnalyzing, setIsAnalyzing] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const [result, setResult] = useState<VisionAnalysisResult | null>(null);

  const analyze = useCallback(async (imageDataUrl: string): Promise<VisionAnalysisResult | null> => {
    setIsAnalyzing(true);
    setError(null);
    setResult(null);

    try {
      const analysisResult = await analyzeWarningLight(imageDataUrl);
      setResult(analysisResult);
      return analysisResult;
    } catch (err) {
      const message = err instanceof Error ? err.message : 'Analysis failed';

      // User-friendly error messages
      if (message.includes('API key')) {
        setError('Service not configured. Please contact support.');
      } else if (message.includes('rate limit')) {
        setError('Too many requests. Please wait a moment and try again.');
      } else if (message.includes('parse')) {
        setError('Could not analyze this image. Please try with a clearer photo.');
      } else {
        setError(message);
      }

      return null;
    } finally {
      setIsAnalyzing(false);
    }
  }, []);

  const reset = useCallback(() => {
    setIsAnalyzing(false);
    setError(null);
    setResult(null);
  }, []);

  return {
    analyze,
    isAnalyzing,
    error,
    result,
    reset,
  };
}
```

Key design decisions:
- Returns both the result and stores it in state for UI access
- User-friendly error messages for common API errors
- Reset function to clear state for new scan
- isAnalyzing flag for loading UI states
