Create src/hooks/useVehicleStorage.ts:

```typescript
import { useCallback, useMemo } from 'react';
import { StoredVehicle, StoredVehicleSchema, Vehicle } from '@/types';

const STORAGE_KEY = 'dash-decoder-vehicle';
const EXPIRY_DAYS = 90;

interface UseVehicleStorageReturn {
  getVehicle: () => StoredVehicle | null;
  saveVehicle: (vehicle: Vehicle) => void;
  clearVehicle: () => void;
  storedVehicle: StoredVehicle | null;
}

export function useVehicleStorage(): UseVehicleStorageReturn {
  const getVehicle = useCallback((): StoredVehicle | null => {
    try {
      const stored = localStorage.getItem(STORAGE_KEY);
      if (!stored) return null;

      const parsed = JSON.parse(stored);
      const validated = StoredVehicleSchema.parse(parsed);

      // Check expiry
      const expiryMs = EXPIRY_DAYS * 24 * 60 * 60 * 1000;
      if (Date.now() - validated.timestamp > expiryMs) {
        localStorage.removeItem(STORAGE_KEY);
        return null;
      }

      return validated;
    } catch {
      // Invalid or corrupted data
      localStorage.removeItem(STORAGE_KEY);
      return null;
    }
  }, []);

  const saveVehicle = useCallback((vehicle: Vehicle) => {
    const storedVehicle: StoredVehicle = {
      ...vehicle,
      timestamp: Date.now(),
    };
    localStorage.setItem(STORAGE_KEY, JSON.stringify(storedVehicle));
  }, []);

  const clearVehicle = useCallback(() => {
    localStorage.removeItem(STORAGE_KEY);
  }, []);

  // Get current stored vehicle (memoized to prevent re-renders)
  const storedVehicle = useMemo(() => getVehicle(), [getVehicle]);

  return {
    getVehicle,
    saveVehicle,
    clearVehicle,
    storedVehicle,
  };
}
```

Key design decisions:
- 90-day expiry per research recommendation
- Uses Zod validation for stored data
- Clears corrupted data instead of crashing
- Memoized storedVehicle for initial check

```tsx
import { useState } from 'react';
import { Button } from '@/components/ui/button';
import { Card, CardContent, CardFooter } from '@/components/ui/card';
import { VehicleSearch } from './VehicleSearch';
import type { VehicleGuess, Vehicle } from '@/types';

interface VehicleCardProps {
  vehicleGuess: VehicleGuess;
  storedVehicle: Vehicle | null;
  onConfirm: (vehicle: Vehicle) => void;
}

export function VehicleCard({
  vehicleGuess,
  storedVehicle,
  onConfirm,
}: VehicleCardProps) {
  const [showSearch, setShowSearch] = useState(false);

  // Check if AI made a guess
  const hasGuess = vehicleGuess.make && vehicleGuess.model && vehicleGuess.year;

  // Use stored vehicle if available, otherwise use AI guess
  const displayVehicle = storedVehicle || (hasGuess ? {
    make: vehicleGuess.make!,
    model: vehicleGuess.model!,
    year: vehicleGuess.year!,
  } : null);

  // If no guess and no stored vehicle, go straight to search
  if (!displayVehicle) {
    return (
      <Card className="w-full max-w-md mx-auto">
        <CardContent className="pt-6">
          <p className="text-sm text-muted-foreground mb-4">
            We couldn't identify your vehicle. Please enter your vehicle details:
          </p>
          <VehicleSearch onSelect={onConfirm} />
        </CardContent>
      </Card>
    );
  }

  if (showSearch) {
    return (
      <Card className="w-full max-w-md mx-auto">
        <CardContent className="pt-6">
          <p className="text-sm text-muted-foreground mb-4">
            Enter your vehicle details:
          </p>
          <VehicleSearch
            initialVehicle={displayVehicle}
            onSelect={onConfirm}
            onCancel={() => setShowSearch(false)}
          />
        </CardContent>
      </Card>
    );
  }

  return (
    <Card className="w-full max-w-md mx-auto">
      <CardContent className="pt-6">
        <p className="text-sm text-muted-foreground mb-2">
          {storedVehicle ? 'Using your saved vehicle:' : 'We think this is a:'}
        </p>
        <p className="text-xl font-semibold">
          {displayVehicle.year} {displayVehicle.make} {displayVehicle.model}
        </p>
        {vehicleGuess.confidence > 0 && vehicleGuess.confidence < 0.7 && !storedVehicle && (
          <p className="text-xs text-muted-foreground mt-1">
            (Confidence: {Math.round(vehicleGuess.confidence * 100)}%)
          </p>
        )}
      </CardContent>

      <CardFooter className="flex gap-2">
        <Button
          onClick={() => onConfirm(displayVehicle)}
          className="flex-1"
          size="lg"
        >
          Yes, that's right
        </Button>
        <Button
          onClick={() => setShowSearch(true)}
          variant="outline"
          size="lg"
        >
          No
        </Button>
      </CardFooter>
    </Card>
  );
}
```

```tsx
import { useState, useCallback, useMemo } from 'react';
import { useForm } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import { Input } from '@/components/ui/input';
import { Label } from '@/components/ui/label';
import { Button } from '@/components/ui/button';
import { VehicleSchema, Vehicle } from '@/types';

// Common vehicle makes for autocomplete
const COMMON_MAKES = [
  'Acura', 'Audi', 'BMW', 'Buick', 'Cadillac', 'Chevrolet', 'Chrysler',
  'Dodge', 'Ford', 'GMC', 'Honda', 'Hyundai', 'Infiniti', 'Jeep', 'Kia',
  'Lexus', 'Lincoln', 'Mazda', 'Mercedes-Benz', 'Nissan', 'Ram', 'Subaru',
  'Tesla', 'Toyota', 'Volkswagen', 'Volvo',
];

interface VehicleSearchProps {
  initialVehicle?: Vehicle;
  onSelect: (vehicle: Vehicle) => void;
  onCancel?: () => void;
}

export function VehicleSearch({
  initialVehicle,
  onSelect,
  onCancel,
}: VehicleSearchProps) {
  const [makeInput, setMakeInput] = useState(initialVehicle?.make || '');
  const [showSuggestions, setShowSuggestions] = useState(false);

  const {
    register,
    handleSubmit,
    setValue,
    formState: { errors },
  } = useForm<Vehicle>({
    resolver: zodResolver(VehicleSchema),
    defaultValues: initialVehicle || {
      make: '',
      model: '',
      year: new Date().getFullYear(),
    },
  });

  const filteredMakes = useMemo(() => {
    if (!makeInput) return COMMON_MAKES;
    const lower = makeInput.toLowerCase();
    return COMMON_MAKES.filter((make) =>
      make.toLowerCase().includes(lower)
    );
  }, [makeInput]);

  const handleMakeSelect = useCallback((make: string) => {
    setMakeInput(make);
    setValue('make', make);
    setShowSuggestions(false);
  }, [setValue]);

  const onSubmit = (data: Vehicle) => {
    onSelect(data);
  };

  return (
    <form onSubmit={handleSubmit(onSubmit)} className="space-y-4">
      {/* Make field with suggestions */}
      <div className="space-y-2">
        <Label htmlFor="make">Make</Label>
        <div className="relative">
          <Input
            id="make"
            placeholder="e.g., Toyota, Honda, Ford"
            value={makeInput}
            onChange={(e) => {
              setMakeInput(e.target.value);
              setValue('make', e.target.value);
            }}
            onFocus={() => setShowSuggestions(true)}
            onBlur={() => {
              // Delay to allow click on suggestion
              setTimeout(() => setShowSuggestions(false), 200);
            }}
            className={errors.make ? 'border-destructive' : ''}
          />
          {showSuggestions && filteredMakes.length > 0 && (
            <ul className="absolute z-10 w-full mt-1 bg-popover border rounded-md shadow-lg max-h-48 overflow-auto">
              {filteredMakes.slice(0, 8).map((make) => (
                <li
                  key={make}
                  className="px-3 py-2 cursor-pointer hover:bg-accent"
                  onMouseDown={() => handleMakeSelect(make)}
                >
                  {make}
                </li>
              ))}
            </ul>
          )}
        </div>
        {errors.make && (
          <p className="text-sm text-destructive">{errors.make.message}</p>
        )}
      </div>

      {/* Model field */}
      <div className="space-y-2">
        <Label htmlFor="model">Model</Label>
        <Input
          id="model"
          placeholder="e.g., Camry, Civic, F-150"
          {...register('model')}
          className={errors.model ? 'border-destructive' : ''}
        />
        {errors.model && (
          <p className="text-sm text-destructive">{errors.model.message}</p>
        )}
      </div>

      {/* Year field */}
      <div className="space-y-2">
        <Label htmlFor="year">Year</Label>
        <Input
          id="year"
          type="number"
          placeholder="e.g., 2020"
          min={1996}
          max={new Date().getFullYear() + 1}
          {...register('year', { valueAsNumber: true })}
          className={errors.year ? 'border-destructive' : ''}
        />
        {errors.year && (
          <p className="text-sm text-destructive">{errors.year.message}</p>
        )}
        <p className="text-xs text-muted-foreground">
          Only OBD-II vehicles (1996 and newer) are supported
        </p>
      </div>

      {/* Buttons */}
      <div className="flex gap-2 pt-2">
        <Button type="submit" className="flex-1">
          Confirm Vehicle
        </Button>
        {onCancel && (
          <Button type="button" variant="outline" onClick={onCancel}>
            Cancel
          </Button>
        )}
      </div>
    </form>
  );
}
```
